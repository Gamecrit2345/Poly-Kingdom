const canvas = document.getElementById('canvas'); const ctx = canvas.getContext('2d'); let resources = { food: 500, wood: 500, stone: 300, gold: 200 }; let buildings = []; let troops = 10; let placingBuilding = null; // Bigger world for panning const GRID_SIZE = 40; const WORLD_COLS = 50; const WORLD_ROWS = 40; let cameraX = 0; let cameraY = 0; let isDragging = false; let lastMouseX = 0; let lastMouseY = 0; // Map tiles (0 = grass, 1 = tree) let map = Array(WORLD_ROWS).fill().map(() => Array(WORLD_COLS).fill(0)); // Add trees for (let i = 0; i < 300; i++) { let x = Math.floor(Math.random() * WORLD_COLS); let y = Math.floor(Math.random() * WORLD_ROWS); map[y][x] = 1; } function updateResources() { document.getElementById('food').textContent = Math.floor(resources.food); document.getElementById('wood').textContent = Math.floor(resources.wood); document.getElementById('stone').textContent = Math.floor(resources.stone); document.getElementById('gold').textContent = Math.floor(resources.gold); } function log(message) { const logDiv = document.getElementById('log'); logDiv.innerHTML += `
${message}
`; logDiv.scrollTop = logDiv.scrollHeight; } function drawMap() { ctx.fillStyle = '#1a3a1a'; ctx.fillRect(0, 0, canvas.width, canvas.height); const startCol = Math.floor(cameraX / GRID_SIZE); const startRow = Math.floor(cameraY / GRID_SIZE); const endCol = Math.ceil((cameraX + canvas.width) / GRID_SIZE); const endRow = Math.ceil((cameraY + canvas.height) / GRID_SIZE); // Grid ctx.strokeStyle = 'rgba(255,255,255,0.08)'; for (let x = startCol; x <= endCol; x++) { const screenX = x * GRID_SIZE - cameraX; ctx.beginPath(); ctx.moveTo(screenX, 0); ctx.lineTo(screenX, canvas.height); ctx.stroke(); } for (let y = startRow; y <= endRow; y++) { const screenY = y * GRID_SIZE - cameraY; ctx.beginPath(); ctx.moveTo(0, screenY); ctx.lineTo(canvas.width, screenY); ctx.stroke(); } // Trees ctx.fillStyle = '#0a2a0a'; for (let y = Math.max(0, startRow); y < Math.min(WORLD_ROWS, endRow); y++) { for (let x = Math.max(0, startCol); x < Math.min(WORLD_COLS, endCol); x++) { if (map[y][x] === 1) { const screenX = x * GRID_SIZE - cameraX; const screenY = y * GRID_SIZE - cameraY; ctx.fillRect(screenX + 5, screenY + 5, GRID_SIZE-10, GRID_SIZE-10); ctx.fillStyle = '#0f5'; ctx.beginPath(); ctx.arc(screenX + GRID_SIZE/2, screenY + GRID_SIZE/3, 14, 0, Math.PI*2); ctx.fill(); ctx.fillStyle = '#0a2a0a'; } } } // Buildings buildings.forEach(b => { const screenX = b.x * GRID_SIZE - cameraX; const screenY = b.y * GRID_SIZE - cameraY; if (screenX < -GRID_SIZE || screenX > canvas.width || screenY < -GRID_SIZE || screenY > canvas.height) return; if (b.type === 'farm') { ctx.fillStyle = '#8b5'; ctx.fillRect(screenX + 4, screenY + 4, GRID_SIZE-8, GRID_SIZE-8); ctx.fillStyle = '#ff0'; ctx.fillRect(screenX + 12, screenY + 15, 10, 10); } else if (b.type === 'barracks') { ctx.fillStyle = '#555'; ctx.fillRect(screenX + 4, screenY + 4, GRID_SIZE-8, GRID_SIZE-8); ctx.fillStyle = '#f44'; ctx.fillRect(screenX + 12, screenY + 12, GRID_SIZE-24, GRID_SIZE-24); } else if (b.type === 'townhall') { ctx.fillStyle = '#cc8'; ctx.fillRect(screenX + 2, screenY + 2, GRID_SIZE-4, GRID_SIZE-4); ctx.fillStyle = '#ff0'; ctx.fillRect(screenX + 15, screenY + 10, GRID_SIZE-30, 10); } }); ctx.fillStyle = '#fff'; ctx.font = 'bold 18px Arial'; ctx.fillText(`👥 ${troops} Warriors`, 20, 35); ctx.font = '12px Arial'; ctx.fillStyle = 'rgba(255,255,255,0.6)'; ctx.fillText(`🖱️ Drag to move map`, canvas.width - 200, 25); } function gameLoop() { drawMap(); requestAnimationFrame(gameLoop); } // === DRAG MAP === canvas.addEventListener('mousedown', (e) => { if (placingBuilding) return; isDragging = true; lastMouseX = e.clientX; lastMouseY = e.clientY; canvas.style.cursor = 'grabbing'; }); canvas.addEventListener('mousemove', (e) => { if (!isDragging) return; const dx = e.clientX - lastMouseX; const dy = e.clientY - lastMouseY; cameraX -= dx; cameraY -= dy; const maxX = WORLD_COLS * GRID_SIZE - canvas.width; const maxY = WORLD_ROWS * GRID_SIZE - canvas.height; cameraX = Math.max(0, Math.min(cameraX, maxX)); cameraY = Math.max(0, Math.min(cameraY, maxY)); lastMouseX = e.clientX; lastMouseY = e.clientY; }); canvas.addEventListener('mouseup', () => { isDragging = false; canvas.style.cursor = 'default'; }); canvas.addEventListener('mouseleave', () => { isDragging = false; canvas.style.cursor = 'default'; }); // === PLACE BUILDING === canvas.addEventListener('click', (e) => { if (!placingBuilding) return; const rect = canvas.getBoundingClientRect(); const mouseX = e.clientX - rect.left; const mouseY = e.clientY - rect.top; const worldX = Math.floor((mouseX + cameraX) / GRID_SIZE); const worldY = Math.floor((mouseY + cameraY) / GRID_SIZE); if (worldX < 0 || worldX >= WORLD_COLS || worldY < 0 || worldY >= WORLD_ROWS) return; const occupied = buildings.some(b => b.x === worldX && b.y === worldY); if (occupied || map[worldY][worldX] === 1) { log("❌ Hindi pwede dito! May nakaharang."); return; } let cost = {}; if (placingBuilding === 'farm') cost = {food: 100}; else if (placingBuilding === 'barracks') cost = {wood: 150}; else if (placingBuilding === 'townhall') cost = {stone: 200}; if (resources.food >= (cost.food || 0) && resources.wood >= (cost.wood || 0) && resources.stone >= (cost.stone || 0)) { resources.food -= cost.food || 0; resources.wood -= cost.wood || 0; resources.stone -= cost.stone || 0; buildings.push({type: placingBuilding, x: worldX, y: worldY}); updateResources(); log(`✅ Napalagay ang ${placingBuilding.toUpperCase()}!`); placingBuilding = null; } else { log("❌ Kulang ang resources!"); } }); function startPlacing(type) { placingBuilding = type; log(`📍 Pumili ng lugar para sa ${type}... (i-drag muna ang mapa kung kailangan)`); } function trainTroops(amount) { const cost = amount * 10; if (resources.food >= cost) { resources.food -= cost; troops += amount; updateResources(); log(`⚔️ Nasasanay ${amount} warriors! Kabuuan: ${troops}`); } else { log("❌ Kulang ang pagkain!"); } } function attack() { if (troops < 5) { log("❌ Hindi sapat ang tropa!"); return; } const win = Math.random() > 0.4; const loot = Math.floor(Math.random() * 120) + 50; if (win) { resources.gold += loot; resources.wood += Math.floor(loot / 2); troops = Math.max(3, troops - Math.floor(Math.random()*3)); log(`🏆 Nanalo! +${loot} Ginto`); } else { troops = Math.max(1, troops - 7); log("💥 Natalo sa laban..."); } updateResources(); } // Resource production setInterval(() => { let foodGain = buildings.filter(b => b.type === 'farm').length * 6; let woodGain = buildings.filter(b => b.type === 'townhall').length * 4; let stoneGain = buildings.filter(b => b.type === 'townhall').length * 3; resources.food += foodGain; resources.wood += woodGain; resources.stone += stoneGain; if (foodGain + woodGain + stoneGain > 0) updateResources(); }, 2500); updateResources(); gameLoop(); log("🌍 **Rise of Kingdoms - Movable Map**"); log("🖱️ I-drag ang mapa para mag-explore!");
